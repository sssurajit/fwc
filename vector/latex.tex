\section{\textbf{Problem}}
Find the distance between the point(0,0) and (36,15).Can you now find the distance between the two towns A and B discussed in Section 7.2
\section{\textbf{Solution}}
The distance betwen the points A and B is given
\begin{equation}
    \Vec A=\myvec{0 & 0}
\end{equation}
\begin{equation}
    \Vec B=\myvec{36 & 15}
\end{equation}

\begin{equation}
   \|\vec {A-B}\|
\end{equation}
where
\begin{equation}
    \vec {A-B}=\myvec{-36\\-15}
\end{equation}
\begin{equation}
    \vec d=\sqrt{\myvec{\vec {A-B}}^T\myvec{\vec {A-B}}}
\end{equation}
\begin{equation}
    \Vec d =\sqrt{\myvec{-36\\-15}{\myvec{-36-15}}}
\end{equation}
\begin{equation}
    \vec d=\sqrt{1296+225}
\end{equation}
\begin{equation}
    \vec d=\sqrt{1521}
\end{equation}
\begin{equation}
    \vec d=39
\end{equation}
